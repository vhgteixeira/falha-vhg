        function deleteScreen() {
            document.body.innerHTML = ''; // Remove all elements inside the body
            const button = document.createElement('button'); // Recreate the button
            button.className = 'rounded-button blink-effect'; // Add the blink effect class
            button.textContent = 'Activating...'; // Change the button text
            button.onclick = deleteScreen; // Set the click event again

            setTimeout(stopBlink, 3000); // Stop blinking after 3 seconds
